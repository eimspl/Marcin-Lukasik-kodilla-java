SELECT ISSUESLISTS.NAME, ISSUES.DESCRIPT FROM ISSUES RIGHT JOIN ISSUESLISTS ON ISSUESLISTS.ID=ISSUESLIST_ID;

SELECT ISSUESLISTS.NAME AS STATUS, ISSUES.DESCRIPT, USERS.LASTNAME FROM ISSUES INNER JOIN ISSUESLISTS ON ISSUESLISTS.ID=ISSUESLIST_ID 
INNER JOIN USERS ON USERS.ID=ISSUES.USER_ID_ASSIGNEDTO;

SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(ISSUES.USER_ID_ASSIGNEDTO) FROM USERS LEFT JOIN ISSUES ON USERS.ID=ISSUES.USER_ID_ASSIGNEDTO GROUP BY ISSUES.USER_ID_ASSIGNEDTO HAVING ISSUES.USER_ID_ASSIGNEDTO>0;
